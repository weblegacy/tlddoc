32  package io.github.weblegacy.tlddoc;
33  
34  import java.io.IOException;
35  import java.nio.file.Path;
36  import java.util.Enumeration;
37  import java.util.jar.JarEntry;
38  import javax.xml.parsers.DocumentBuilder;
39  import javax.xml.transform.TransformerException;
40  import javax.xml.transform.TransformerFactoryConfigurationError;
41  import org.w3c.dom.Document;
42  import org.w3c.dom.Element;
43  import org.xml.sax.SAXException;
44  
45  /**
46   * Implicit Tag Library for a directory of tag files that is encapsulated in a WAR file.
47   *
48   * @author mroth
49   */
50  public class WarTagDirImplicitTagLibrary extends WarJarTagLibrary {
51  
52      /**
53       * Creates a new instance of {@link WarTagDirImplicitTagLibrary}.
54       *
55       * @param war WAR file that contains this tag library
56       * @param dir directory containing the tag files
57       */
58      public WarTagDirImplicitTagLibrary(Path war, String dir) {
59          super(war, dir);
60      }
61  
62      /**
63       * {@inheritDoc}
64       */
65      @Override
66      public Document getTldDocument(DocumentBuilder documentBuilder) throws IOException,
67              SAXException, TransformerFactoryConfigurationError, TransformerException {
68  
69          Document result = documentBuilder.newDocument();
70  
71          // Determine path from root of web application:
72          String path = getEntry();
73          if (!path.endsWith("/")) {
74              path += "/";
75          }
76  
77          Element taglibElement = TagDirImplicitTagLibrary.createRootTaglibNode(result,
78                  '/' + path);
79  
80          // According to the JSP 2.0 specification:
81          // A <tag-file> element is considered to exist for each tag file in
82          // this directory, with the following sub-elements:
83          //    - The <name> for each is the filename of the tag file,
84          //      without the .tag extension.
85          //    - The <path> for each is the path of the tag file, relative
86          //      to the root of the web application.
87          ensureOpen();
88          Enumeration<JarEntry> entries = getWarJarFile().entries();
89          while (entries.hasMoreElements()) {
90              JarEntry warEntry = entries.nextElement();
91              String entryName = warEntry.getName();
92              if (!warEntry.isDirectory() && entryName.startsWith(path)) {
93                  String relativeName = entryName.replace(Utils.DEFAULT_SEPARATOR, "/");
94                  relativeName = relativeName.substring(path.length());
95                  if (relativeName.indexOf('/') == -1 && Utils.isTag(relativeName)) {
96                      // We're not in a subdirectory and file and ends with .tag or .tagx.
97                      String tagName = relativeName.substring(0, relativeName.lastIndexOf('.'));
98                      String tagPath = "/" + entryName;
99  
100                     TagDirImplicitTagLibrary.createTagEntry(result, tagName, tagPath,
101                             taglibElement);
102                 }
103             }
104         }
105 
106         return TagDirImplicitTagLibrary.recreateDocument(documentBuilder, result);
107     }
108 }
