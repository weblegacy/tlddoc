55  /**
56   * Implicit Tag Library for a directory of tag files.
57   *
58   * @author mroth
59   */
60  public class TagDirImplicitTagLibrary implements TagLibrary {
61  
62      /**
63       * The directory containing the tag files.
64       */
65      private final Path dir;
66  
67      /**
68       * Creates a new instance of {@link TagDirImplicitTagLibrary}.
69       *
70       * @param dir directory containing the tag files
71       */
72      public TagDirImplicitTagLibrary(Path dir) {
73          this.dir = dir;
74      }
75  
76      /**
77       * {@inheritDoc}
78       */
79      @Override
80      public String getPathDescription() {
81          return dir.toAbsolutePath().toString();
82      }
83  
84      /**
85       * {@inheritDoc}
86       */
87      @Override
88      public InputStream getResource(String path) throws IOException {
89          return Utils.backtrackPath(dir, path);
90      }
91  
92      /**
93       * {@inheritDoc}
94       */
95      @Override
96      public Document getTldDocument(DocumentBuilder documentBuilder) throws IOException,
97              SAXException, TransformerFactoryConfigurationError, TransformerException {
98  
99          Document result = documentBuilder.newDocument();
100 
101         // Determine path from root of web application (this is somewhat of
102         // a guess):
103         String path = Utils.pathString(dir);
104         int index = path.indexOf("/WEB-INF/");
105         if (index != -1) {
106             path = path.substring(index);
107         } else {
108             path = "unknown";
109         }
110         if (!path.endsWith("/")) {
111             path += "/";
112         }
113 
114         // Create root taglib node:
115         Element taglibElement = createRootTaglibNode(result, path);
116 
117         // According to the JSP 2.0 specification:
118         // A <tag-file> element is considered to exist for each tag file in
119         // this directory, with the following sub-elements:
120         //    - The <name> for each is the filename of the tag file,
121         //      without the .tag extension.
122         //    - The <path> for each is the path of the tag file, relative
123         //      to the root of the web application.
124         final String p = path;
125         try (DirectoryStream<Path> files = Files.newDirectoryStream(dir, Utils::isTag)) {
126             for (Path file : files) {
127                 final Path fn = file.getFileName();
128                 if (fn == null) {
129                     continue;
130                 }
131 
132                 final String fileName = fn.toString();
133                 final String tagName = fileName.substring(0, fileName.lastIndexOf('.'));
134                 final String tagPath = p + fileName;
135 
136                 createTagEntry(result, tagName, tagPath, taglibElement);
137             }
138         }
139 
140         return recreateDocument(documentBuilder, result);
141     }
142 
143     /**
144      * Creates an implicit tag library root node, with default values. Shared by
145      * WarTagDirImplicitTagLibrary.
146      *
147      * @param result XML-document to add new tag-element
148      * @param path   path to the TLD Files
149      *
150      * @return new created tag library root node
151      */
152     static Element createRootTaglibNode(Document result, String path) {
153         Element taglibElement = result.createElementNS(Constants.NS_JAKARTAEE, "taglib");
154         // JDK 1.4 does not add xmlns for some reason - add it manually:
155         taglibElement.setAttributeNS("http://www.w3.org/2000/xmlns/",
156                 "xmlns", Constants.NS_JAKARTAEE);
157         taglibElement.setAttributeNS("http://www.w3.org/2000/xmlns/",
158                 "xmlns:xsi", "http://www.w3.org/2001/XMLSchema-instance");
159         taglibElement.setAttributeNS(
160                 "http://www.w3.org/2001/XMLSchema-instance",
161                 "xsi:schemaLocation",
162                 Constants.NS_JAKARTAEE
163                 + " https://jakarta.ee/xml/ns/jakartaee/web-jsptaglibrary_3_0.xsd");
164         taglibElement.setAttribute("version", "3.0");
165         result.appendChild(taglibElement);
166 
167         // Add <description>
168         Element descriptionElement = result.createElement("description");
169         descriptionElement.appendChild(result.createTextNode(
170                 "Implicit tag library for tag file directory " + path));
171         taglibElement.appendChild(descriptionElement);
172 
173         // Add <tlib-version> of 1.0
174         Element tlibVersionElement = result.createElement("tlib-version");
175         tlibVersionElement.appendChild(result.createTextNode("1.0"));
176         taglibElement.appendChild(tlibVersionElement);
177 
178         // According to the JSP 2.0 specification, <short-name> is derived
179         // from the directory name. If the directory is /WEB-INF/tags/, the
180         // short name is simply tags. Otherwise, the full directory path
181         // (relative to the web application) is taken, minus the
182         // /WEB-INF/tags/ prefix. Then, all / characters are replaced
183         // with -, which yields the short name. Note that short names are
184         // not guaranteed to be unique.
185         String shortName;
186         switch (path) {
187             case "unknown":
188                 shortName = path;
189                 break;
190             case "/WEB-INF/tags":
191             case "/WEB-INF/tags/":
192                 shortName = "tags";
193                 break;
194             default:
195                 shortName = path;
196                 if (shortName.startsWith("/WEB-INF/tags")) {
197                     shortName = shortName.substring("/WEB-INF/tags".length());
198                 }
199                 if (shortName.startsWith("/")) {
200                     shortName = shortName.substring(1);
201                 }
202                 if (shortName.endsWith("/")) {
203                     shortName = shortName.substring(0, shortName.length() - 1);
204                 }
205                 shortName = shortName.replace('/', '-');
206                 break;
207         }
208         Element shortNameElement = result.createElement("short-name");
209         shortNameElement.appendChild(result.createTextNode(shortName));
210         taglibElement.appendChild(shortNameElement);
211 
212         Element uriElement = result.createElement("uri");
213         uriElement.appendChild(result.createTextNode(path));
214         taglibElement.appendChild(uriElement);
215 
216         return taglibElement;
217     }

219     /**
220      * Creates an tag-entry into tag library root node. Shared by WarTagDirImplicitTagLibrary.
221      *
222      * @param result        XML-document to add new tag-element
223      * @param tagName       name of the tag-entry
224      * @param tagPath       path of the tag-entry
225      * @param taglibElement tag library root node
226      *
227      * @throws DOMException if an XML error has occurred
228      */
229     static void createTagEntry(final Document result, final String tagName, final String tagPath,
230             Element taglibElement) throws DOMException {
231 
232         final Element tagFileElement = result.createElement("tag-file");
233         final Element nameElement = result.createElement("name");
234         nameElement.appendChild(result.createTextNode(tagName));
235         tagFileElement.appendChild(nameElement);
236 
237         final Element pathElement = result.createElement("path");
238         pathElement.appendChild(result.createTextNode(tagPath));
239         tagFileElement.appendChild(pathElement);
240         taglibElement.appendChild(tagFileElement);
241     }

243     /**
244      * Recreates the XML document. JDK 1.4 does not correctly import the node into the tree, so
245      * simulate reading this entry from a file. There might be a better / more efficient way to do
246      * this, but this works.
247      *
248      * @param documentBuilder {@code DocumentBuilder} to obtain DOM Document for creating an
249      *                        XML-document.
250      * @param result          XML-document to recreate
251      *
252      * @return the recreated XML-document
253      *
254      * @throws IOException                          if an I/O error has occurred
255      * @throws SAXException                         If any parse errors occur.
256      * @throws TransformerFactoryConfigurationError Thrown in case of {@linkplain
257      * java.util.ServiceConfigurationError service configuration error} or if the implementation is
258      *                                              not available or cannot be instantiated.
259      * @throws TransformerException                 If an unrecoverable error occurs during the
260      *                                              course of the transformation.
261      */
262     static Document recreateDocument(final DocumentBuilder documentBuilder, final Document result)
263             throws IOException, TransformerFactoryConfigurationError, TransformerException,
264             SAXException {
265 
266         final StringWriter buffer = new StringWriter();
267         Transformer transformer = TransformerFactory.newInstance().newTransformer();
268         transformer.transform(new DOMSource(result), new StreamResult(buffer));
269 
270         return documentBuilder.parse(new InputSource(new StringReader(buffer.toString())));
271     }

278         // Nothing to do
279     }
