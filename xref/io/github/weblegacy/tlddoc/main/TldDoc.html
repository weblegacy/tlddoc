93      /**
94       * The main-entry-point for TldDoc.
95       *
96       * @param args Arguments from command line
97       */
98      public static void main(String[] args) {
99          TldDocGenerator generator = new TldDocGenerator();
100 
101         try {
102             initGenerator(args, generator);
103             generator.generate();
104         } catch (IOException | GeneratorException e) {
105             e.printStackTrace(System.err);
106             System.exit(1);
107         }
108     }
109 
110     private static void initGenerator(final String[] args, final TldDocGenerator generator)
111             throws IOException {
112 
113         Iterator<String> iter = Arrays.asList(args).iterator();
114         boolean atLeastOneTld = false;
115 
116         try {
117             while (iter.hasNext()) {
118                 String arg = iter.next();
119 
120                 switch (arg) {
121                     case "-xslt":
122                         arg = iter.next();
123                         generator.setXsltDirectory(Paths.get(arg));
124                         break;
125                     case "-d":
126                         arg = iter.next();
127                         generator.setOutputDirectory(Paths.get(arg));
128                         break;
129                     case "-help":
130                         usage(null);
131                         break;
132                     case "-q":
133                         generator.setQuiet(true);
134                         break;
135                     case "-v":
136                         generator.setVerbose(true);
137                         break;
138                     case "-doctitle":
139                         arg = iter.next();
140                         generator.setDocTitle(arg);
141                         break;
142                     case "-windowtitle":
143                         arg = iter.next();
144                         generator.setWindowTitle(arg);
145                         break;
146                     case "-webapp":
147                         atLeastOneTld |= addPath(iter, generator::addWebApp, "Web app");
148                         break;
149                     case "-jar":
150                         atLeastOneTld |= addPath(iter, generator::addJar, "JAR");
151                         break;
152                     case "-tagdir":
153                         atLeastOneTld |= addPath(iter, generator::addTagDir, "Tag Directory");
154                         break;
155                     default:
156                         Path f = Paths.get(arg);
157                         if (Files.exists(f)) {
158                             String fn = Utils.getLowerFileName(f);
159                             if (fn.endsWith(".tld")) {
160                                 // If the path is a file that ends in .tld,
161                                 // process an individual TLD file.
162                                 generator.addTld(f);
163                                 atLeastOneTld = true;
164                             } else if (fn.endsWith(".jar")) {
165                                 // If the path is a file that ends in .jar,
166                                 // process a tag library JAR file.
167                                 generator.addJar(f);
168                                 atLeastOneTld = true;
169                             } else if (fn.endsWith(".war")) {
170                                 // If the path is a file that ends in .war,
171                                 // process all tag libraries in this web app.
172                                 generator.addWar(f);
173                                 atLeastOneTld = true;
174                             } else {
175                                 boolean foundWebInf = false;
176                                 if (Files.isDirectory(f)) {
177                                     // If the path is a directory that includes
178                                     // /WEB-INF/tags process the implicit tag library
179                                     // for the given directory of tag files.
180                                     if (Utils.pathContains(f, "/WEB-INF/tags")) {
181                                         generator.addTagDir(f);
182                                         foundWebInf = true;
183                                     } else {
184                                         if (f.endsWith("WEB-INF")) {
185                                             foundWebInf = true;
186                                         } else {
187                                             try (DirectoryStream<Path> files = Files
188                                                     .newDirectoryStream(f, Utils::isWebInf)) {
189 
190                                                 foundWebInf = files.iterator().hasNext();
191                                             }
192                                         }
193 
194                                         if (foundWebInf) {
195                                             generator.addWebApp(f);
196                                         }
197                                     }
198                                 }
199                                 if (foundWebInf) {
200                                     atLeastOneTld = true;
201                                 } else {
202                                     usage("Cannot determine tag library "
203                                             + "type for " + f.toAbsolutePath());
204                                 }
205                             }
206                         } else {
207                             usage("File/directory not found: " + arg);
208                         }
209                         break;
210                 }
211             }
212             if (!atLeastOneTld) {
213                 usage("Please specify at least one TLD file.");
214             }
215         } catch (NoSuchElementException e) {
216             usage("Invalid Syntax.");
217         }
218     }
219 
220     private static boolean addPath(final Iterator<String> iter, final Consumer<Path> addDirFunction,
221             final String type) throws IOException {
222 
223         final String arg = iter.next();
224         final Path dir = Paths.get(arg);
225         if (Files.exists(dir)) {
226             addDirFunction.accept(dir);
227             return true;
228         }
229         usage(type + " not found: " + arg);
230         return false;
231     }
232 
233     private static void usage(String message) {
234         if (message != null) {
235             System.out.println("Error: " + message);
236         }
237         System.out.println(USAGE);
238         System.exit(0);
239     }
